# 12. 综合案例与复习路线

## 12.1 综合项目应该练什么

学完基础语法后，不要只刷零散语法题，应该做小项目，把知识串起来。

一个合格的入门综合项目最好覆盖：

1. 输入输出。
2. 判断和循环。
3. 函数拆分。
4. 列表、字典等容器。
5. 文件持久化。
6. 异常处理。
7. 模块拆分。
8. 可选：面向对象。
9. 可选：数据可视化。

本章整理两个方向：

1. 学生管理系统。
2. 销售数据统计和可视化。

## 12.2 项目一：学生管理系统

### 12.2.1 需求

实现一个命令行学生管理系统：

1. 添加学生。
2. 删除学生。
3. 修改学生。
4. 查询学生。
5. 显示所有学生。
6. 保存到 JSON 文件。
7. 程序启动时自动读取 JSON 文件。

学生字段：

```text
学号 student_id
姓名 name
年龄 age
成绩 score
```

## 12.3 函数版学生管理系统

```python
import json
from pathlib import Path

DATA_FILE = Path("students.json")


def load_students():
    if not DATA_FILE.exists():
        return []

    try:
        with open(DATA_FILE, "r", encoding="utf-8") as file:
            return json.load(file)
    except json.JSONDecodeError:
        print("数据文件格式错误，已使用空数据启动")
        return []


def save_students(students):
    with open(DATA_FILE, "w", encoding="utf-8") as file:
        json.dump(students, file, ensure_ascii=False, indent=2)


def show_menu():
    print("====== 学生管理系统 ======")
    print("1. 添加学生")
    print("2. 删除学生")
    print("3. 修改学生")
    print("4. 查询学生")
    print("5. 显示所有学生")
    print("0. 退出系统")


def find_student(students, student_id):
    for student in students:
        if student["student_id"] == student_id:
            return student
    return None


def input_int(prompt):
    while True:
        try:
            return int(input(prompt))
        except ValueError:
            print("请输入整数")


def input_float(prompt):
    while True:
        try:
            return float(input(prompt))
        except ValueError:
            print("请输入数字")


def add_student(students):
    student_id = input("请输入学号:")

    if find_student(students, student_id):
        print("学号已存在")
        return

    name = input("请输入姓名:")
    age = input_int("请输入年龄:")
    score = input_float("请输入成绩:")

    students.append({
        "student_id": student_id,
        "name": name,
        "age": age,
        "score": score,
    })
    save_students(students)
    print("添加成功")


def delete_student(students):
    student_id = input("请输入要删除的学号:")
    student = find_student(students, student_id)

    if student is None:
        print("学生不存在")
        return

    students.remove(student)
    save_students(students)
    print("删除成功")


def update_student(students):
    student_id = input("请输入要修改的学号:")
    student = find_student(students, student_id)

    if student is None:
        print("学生不存在")
        return

    name = input("请输入新姓名:")
    age = input_int("请输入新年龄:")
    score = input_float("请输入新成绩:")

    student["name"] = name
    student["age"] = age
    student["score"] = score

    save_students(students)
    print("修改成功")


def query_student(students):
    student_id = input("请输入要查询的学号:")
    student = find_student(students, student_id)

    if student is None:
        print("学生不存在")
        return

    print_student(student)


def print_student(student):
    print(
        f"{student['student_id']}\t"
        f"{student['name']}\t"
        f"{student['age']}\t"
        f"{student['score']}"
    )


def show_all_students(students):
    if not students:
        print("暂无学生")
        return

    print("学号\t姓名\t年龄\t成绩")
    for student in students:
        print_student(student)


def main():
    students = load_students()

    while True:
        show_menu()
        choice = input("请输入操作:")

        if choice == "1":
            add_student(students)
        elif choice == "2":
            delete_student(students)
        elif choice == "3":
            update_student(students)
        elif choice == "4":
            query_student(students)
        elif choice == "5":
            show_all_students(students)
        elif choice == "0":
            print("退出系统")
            break
        else:
            print("输入错误")


if __name__ == "__main__":
    main()
```

这个项目串联了：

1. 菜单循环。
2. 函数封装。
3. 列表字典。
4. JSON 读写。
5. 异常处理。
6. `if __name__ == "__main__"`。

## 12.4 面向对象版设计思路

可以进一步拆成两个类：

1. `Student`：表示单个学生。
2. `StudentManager`：负责学生增删改查和文件保存。

示意代码：

```python
from dataclasses import dataclass


@dataclass
class Student:
    student_id: str
    name: str
    age: int
    score: float

    def to_dict(self):
        return {
            "student_id": self.student_id,
            "name": self.name,
            "age": self.age,
            "score": self.score,
        }

    @classmethod
    def from_dict(cls, data):
        return cls(
            data["student_id"],
            data["name"],
            data["age"],
            data["score"],
        )
```

管理器示意：

```python
class StudentManager:
    def __init__(self):
        self.students = []

    def find_by_id(self, student_id):
        for student in self.students:
            if student.student_id == student_id:
                return student
        return None

    def add(self, student):
        if self.find_by_id(student.student_id):
            return False
        self.students.append(student)
        return True
```

如果项目逻辑越来越多，面向对象版会比纯函数版更容易扩展。

category_total = {

## 12.7 绘制时间线柱状图

时间线适合展示不同日期、月份、年份的数据变化。

```python
from pyecharts import options as opts
from pyecharts.charts import Bar, Timeline

data = {
    "2024-10-01": {"手机": 12000, "电脑": 18000},
    "2024-10-02": {"手机": 15000, "电脑": 16000},
    "2024-10-03": {"手机": 17000, "电脑": 19000},
}

timeline = Timeline()

for date, category_data in data.items():
    bar = Bar()
    bar.add_xaxis(list(category_data.keys()))
    bar.add_yaxis("销售额", list(category_data.values()))
    bar.set_global_opts(title_opts=opts.TitleOpts(title=f"{date}销售额"))
    timeline.add(bar, date)

timeline.render("timeline_sales.html")
```

## 12.8 绘制地图

地图适合展示地区数据。

```python
from pyecharts import options as opts
from pyecharts.charts import Map

data = [
    ("北京", 120),
    ("上海", 150),
    ("广东", 200),
    ("浙江", 180),
]

map_chart = Map()
map_chart.add("销售额", data, "china")
map_chart.set_global_opts(
    title_opts=opts.TitleOpts(title="地区销售额"),
    visualmap_opts=opts.VisualMapOpts(max_=200),
)
map_chart.render("map_sales.html")
```

地图数据的地区名称要和 pyecharts 内置地图名称匹配，否则可能显示不出来。

## 12.13 如何读报错

报错通常从下往上看最后几行。

示例：

```text
ValueError: invalid literal for int() with base 10: 'abc'
```

说明：

1. 异常类型是 `ValueError`。
2. 原因是 `int()` 无法把 `'abc'` 转成整数。
3. 应该检查输入内容或增加异常处理。

再看文件名和行号：

```text
File "main.py", line 10, in <module>
```

表示错误发生在 `main.py` 第 10 行。
